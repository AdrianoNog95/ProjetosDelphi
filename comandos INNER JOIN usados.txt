Juntar Tudo
SELECT R.ID, R.VALOR_PONTUACAO, R.DATA_PONTUACAO, R.CATEGORIA, R.ID_USU, U.NOME_USUARIO FROM RANKING AS R INNER JOIN USUARIO AS U ON R.ID_USU = U.ID ORDER BY R.VALOR_PONTUACAO DESC

Juntar e filtrar para o brasileirão
SELECT R.ID, R.VALOR_PONTUACAO, R.DATA_PONTUACAO, R.CATEGORIA, R.ID_USU, U.NOME_USUARIO FROM RANKING AS R INNER JOIN USUARIO AS U ON R.ID_USU = U.ID WHERE R.CATEGORIA = ''CAMPEONATO BRASILEIRO 2021'' ORDER BY R.VALOR_PONTUACAO DESC
